type typed_name = string * Ty.ty
type typed_binop = Ast.bin_op * Ty.ty
type texpr =
  1. | TConst of Ast.const * Ty.ty
    (*

    Typed expression for the constant

    *)
  2. | TVar of string * Ty.ty
    (*

    Typed expression for the variables

    *)
  3. | TBinop of typed_binop * texpr * texpr
    (*

    Typed expression for the binary operations

    *)
  4. | TApp of texpr * texpr * Ty.ty
    (*

    Typed expression for the function application to the arguments

    *)
  5. | TIfThenElse of texpr * texpr * texpr * Ty.ty
    (*

    Typed expression for condition statement

    *)
  6. | TLetIn of tpattern * texpr * texpr
    (*

    Typed expression for let in declaration

    *)
  7. | TLetRecIn of typed_name * texpr * texpr
    (*

    Typed expression for let rec in declaration

    *)
  8. | TFun of tpattern * texpr * Ty.ty
    (*

    Typed expression for function

    *)
  9. | TTuple of texpr list * Ty.ty
    (*

    Typed expression for the tuples

    *)
