type arg =
  1. | Used of string
    (*

    Represent usual argument

    *)
  2. | Unused
    (*

    Represent unused argument (wildcard for example)

    *)
type immexpr =
  1. | ImmNum of int
    (*

    ..., -1, 0, 1, ...

    *)
  2. | ImmBool of bool
    (*

    true, false

    *)
  3. | ImmId of string
    (*

    identifiers or variables

    *)
type cexpr =
  1. | CBinOp of Ast.bin_op * immexpr * immexpr
    (*

    Binary operation

    *)
  2. | CApp of immexpr * immexpr list
    (*

    Apply function to its arguments

    *)
  3. | CTuple of immexpr list
    (*

    (1, 2, a, b)

    *)
  4. | CTake of immexpr * int
    (*

    Take(tuple, 0)

    *)
  5. | CMakeClosure of immexpr * int * int * immexpr list
    (*

    Used in partial application

    *)
  6. | CAddArgsToClosure of immexpr * int * immexpr list
    (*

    Add arguments to closure

    *)
  7. | CIfThenElse of immexpr * aexpr * aexpr
    (*

    if immexpr then aexpr2 else aexpr3

    *)
  8. | CImmExpr of immexpr
    (*

    immexpr

    *)
and aexpr =
  1. | ALet of string * cexpr * aexpr
    (*

    let name = cexpr in aexpr

    *)
  2. | ACEexpr of cexpr
    (*

    cexpr

    *)
type anfexpr =
  1. | AnfLetFun of string * arg list * aexpr
    (*

    let name arg1, arg2, ..., argn = aexpr. It's possible to have zero args

    *)

Anf binding type (top level declarations)

type anfstatements = anfexpr list

Statements type (list of top level declarations)
