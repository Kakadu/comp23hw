type expr =
  1. | EConst of const
    (*

    An expression for the constant

    *)
  2. | EVar of string
    (*

    An expression for the variables

    *)
  3. | ETuple of expr list
    (*

    An expression for the tuples

    *)
  4. | EBinop of bin_op * expr * expr
    (*

    An expression for the binary operations: +, -, *, / ...

    *)
  5. | EApp of expr * expr
    (*

    An expression for the function application to the arguments

    *)
  6. | EIfThenElse of expr * expr * expr
    (*

    An expression for condition statement: if expr then expr else expr

    *)
  7. | ELetIn of pattern * expr * expr
    (*

    An expression for let in declaration: let id = expr in expr

    *)
  8. | ELetRecIn of string * expr * expr
    (*

    An expression for let rec in declaration: let rec id = expr in expr

    *)
  9. | EFun of pattern * expr
    (*

    An expression for function: fun pattern -> expr

    *)
