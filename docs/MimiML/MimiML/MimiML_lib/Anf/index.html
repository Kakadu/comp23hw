type immexpr =
  1. | ImmValue of string
  2. | ImmNum of int
  3. | ImmString of string
  4. | ImmBool of bool
  5. | ImmUnit
type cexpr =
  1. | CImm of immexpr
  2. | CApp of immexpr * immexpr
  3. | CIfElse of immexpr * ablock * ablock
and alet = string * cexpr
and ablock = immexpr * alet list
type aprogram = ablock
type afun = string * string list * ablock
val anf : Ast.expr -> afun list * aprogram
