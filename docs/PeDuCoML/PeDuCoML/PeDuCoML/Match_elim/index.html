type match_free_expr =
  1. | MFLiteral of Ast.literal
  2. | MFBinaryOperation of Ast.binary_operator * match_free_expr * match_free_expr
  3. | MFUnaryOperation of Ast.unary_operator * match_free_expr
  4. | MFApplication of match_free_expr * match_free_expr
  5. | MFIdentifier of Ast.id
  6. | MFList of match_free_expr list
  7. | MFConstructList of match_free_expr * match_free_expr
  8. | MFLetIn of match_free_let * match_free_let list * match_free_expr
  9. | MFTuple of match_free_expr * match_free_expr * match_free_expr list
  10. | MFIf of match_free_expr * match_free_expr * match_free_expr
and match_free_let = Ast.id * match_free_expr
type match_free_decl =

  2. | MFRecursiveDeclaration of Ast.id * Ast.id list * match_free_expr
