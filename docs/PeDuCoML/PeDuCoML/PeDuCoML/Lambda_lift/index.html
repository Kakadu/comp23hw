type cf_expr =
  1. | CFLiteral of Ast.literal
  2. | CFBinaryOperation of Ast.binary_operator * cf_expr * cf_expr
  3. | CFUnaryOperation of Ast.unary_operator * cf_expr
  4. | CFApplication of cf_expr * cf_expr
  5. | CFIdentifier of Ast.id
  6. | CFList of cf_expr list
  7. | CFConstructList of cf_expr * cf_expr
  8. | CFLetIn of cf_let * cf_let list * cf_expr
  9. | CFTuple of cf_expr * cf_expr * cf_expr list
  10. | CFMatchWith of cf_expr * Ast.pattern * cf_expr * (Ast.pattern * cf_expr) list
  11. | CFIf of cf_expr * cf_expr * cf_expr
and cf_let = Ast.id * cf_expr
type cf_decl =
  1. | CFDeclaration of Ast.id * Ast.id list * cf_expr
  2. | CFRecursiveDeclaration of Ast.id * Ast.id list * cf_expr
val run_lambda_lifting : Ast.declaration list -> cf_decl list
