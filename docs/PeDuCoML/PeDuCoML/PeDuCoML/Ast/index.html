type pattern =
  1. | PLiteral of literal
    (*

    true

    *)
  2. | PWildcard
    (*

    _

    *)
  3. | PTuple of pattern * pattern * pattern list
    (*

    (1, 2)

    *)
  4. | PList of pattern list
    (*

    a; b; c

    *)
  5. | PConstructList of pattern * pattern
    (*

    a :: b; c

    *)
  6. | PIdentifier of id
    (*

    cool_variable

    *)
type expression =
  1. | ELiteral of literal
    (*

    123

    *)
  2. | EBinaryOperation of binary_operator * expression * expression
    (*

    1 + 3

    *)
  3. | EUnaryOperation of unary_operator * expression
    (*

    -(1 + 3)

    *)
  4. | EApplication of expression * expression
    (*

    f x

    *)
  5. | EIdentifier of id
    (*

    x

    *)
  6. | EFun of pattern * pattern list * expression
    (*

    fun x y -> x + y

    *)
  7. | EList of expression list
    (*

    1; 2; 3

    *)
  8. | EConstructList of expression * expression
    (*

    1 :: 2; 3

    *)
  9. | ETuple of expression * expression * expression list
    (*

    (1, "Vasya Pupkin", '\n')

    *)
  10. | ELetIn of declaration * declaration list * expression
    (*

    let x = 1 and y = 2 in x + y

    *)
  11. | EIf of expression * expression * expression
    (*

    if true then 1 else 0

    *)
  12. | EMatchWith of expression * pattern * expression * (pattern * expression) list
    (*

    match x with _ -> x

    *)
and declaration =
  1. | DDeclaration of id * pattern list * expression
    (*

    let add x y = x + y

    *)
  2. | DRecursiveDeclaration of id * pattern list * expression
    (*

    let rec factorial n = n * factorial (n - 1)

    *)
