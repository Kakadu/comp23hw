type unique_id =
  1. | AnfId of int
  2. | GlobalScopeId of string
type imm_expr =
  1. | ImmInt of int
  2. | ImmString of string
  3. | ImmChar of char
  4. | ImmBool of bool
  5. | ImmList of imm_expr list
  6. | ImmTuple of imm_expr list
  7. | ImmId of unique_id
type cexpr =
  1. | CBinaryOperation of Ast.binary_operator * imm_expr * imm_expr
  2. | CUnaryOperation of Ast.unary_operator * imm_expr
  3. | CApplication of imm_expr * imm_expr
  4. | CIf of imm_expr * aexpr * aexpr
  5. | CConstructList of imm_expr * imm_expr
  6. | CImm of imm_expr
and aexpr =
  1. | ALet of unique_id * cexpr * aexpr
  2. | ACExpr of cexpr
type global_scope_function = string * unique_id list * aexpr
val run_anf_conversion : Match_elim.match_free_decl list -> global_scope_function list
