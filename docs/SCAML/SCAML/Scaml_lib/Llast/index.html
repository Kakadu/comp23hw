Ast without anonymous functions, arguments are listed, the recursiveness of nested binding has been removed, since it makes no sense to store it for a variable

type llexpr =
  1. | LLConst of Ast.const
    (*

    constants such as integers, boolean, unit

    *)
  2. | LLBinOp of Ast.bin_op * llexpr * llexpr
    (*

    binary operations such as add, multiply, equality checking etc.

    *)
  3. | LLVar of Ast.id
    (*

    identifier for variables

    *)
  4. | LLIf of llexpr * llexpr * llexpr
    (*

    conditionals

    *)
  5. | LLApp of llexpr * llexpr
    (*

    applying a function to an argument

    *)
  6. | LLLetIn of Ast.id * llexpr * llexpr
    (*

    binding to a variable

    *)
val pp_llexpr : Ppx_deriving_runtime.Format.formatter -> llexpr -> Ppx_deriving_runtime.unit
val show_llexpr : llexpr -> Ppx_deriving_runtime.string
