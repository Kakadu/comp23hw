type immexpr =
  1. | ImmInt of int
  2. | ImmBool of bool
  3. | ImmIdentifier of string
type aexpr =
  1. | ALet of string * cexpr * aexpr
  2. | ACExpr of cexpr
and cexpr =
  1. | CImmExpr of immexpr
  2. | CUnaryOp of Ast.un_op * immexpr
  3. | CBinaryOp of Ast.bin_op * immexpr * immexpr
  4. | CApp of immexpr * immexpr
  5. | CIf of immexpr * aexpr * aexpr
type pexpr =
  1. | PImmExpr of immexpr
  2. | PImmWild
type abinding =
  1. | ABind of string * pexpr list * aexpr
val anf_program : Ast.prog -> abinding list
