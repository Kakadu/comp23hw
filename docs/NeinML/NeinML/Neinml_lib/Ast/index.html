type 'a expression =
  1. | BinOp of 'a expression * 'a expression * binop * 'a
    (*

    x * y

    *)
  2. | LetIn of name * 'a expression * 'a expression * 'a
    (*

    let func = ... in ...

    *)
  3. | RecLetIn of name * 'a expression * 'a expression * 'a
    (*

    let rec func = ... in ...

    *)
  4. | IfThenElse of 'a expression * 'a expression * 'a expression * 'a
    (*

    if condition then expr1 else expr2

    *)
  5. | Func of name * 'a expression * 'a
    (*

    ast expression for defining functions with currying

    *)
  6. | Apply of 'a expression * 'a expression * 'a
    (*

    func arg1 arg2 ...

    *)
  7. | Variable of name * 'a
    (*

    var

    *)
  8. | Value of const * 'a
    (*

    value (const)

    *)
