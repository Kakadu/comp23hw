type dispatch = {
  1. func_call : dispatch -> unit Ast.expression Angstrom.t;
    (*

    Parser for function call

    *)
  2. parse_lam : dispatch -> string -> unit Ast.expression Angstrom.t;
    (*

    Parser for anonymous functions (fun ... -> ...)

    *)
  3. parse_if : dispatch -> string -> unit Ast.expression Angstrom.t;
  4. arithmetical : dispatch -> string -> unit Ast.expression Angstrom.t;
    (*

    Parser for arithmetic operations

    *)
  5. parse_priority : dispatch -> string -> unit Ast.expression Angstrom.t;
    (*

    Parser for only arithmetic operations *, /, %

    *)
  6. logical : dispatch -> string -> unit Ast.expression Angstrom.t;
    (*

    Parser for logic operations without && and || operators

    *)
  7. parse_and : dispatch -> string -> unit Ast.expression Angstrom.t;
    (*

    Parser for logic operations with && operator

    *)
  8. logical_sequence : dispatch -> string -> unit Ast.expression Angstrom.t;
    (*

    Parser for logic operations with && and || operators

    *)
  9. bracket : dispatch -> unit Ast.expression Angstrom.t;
    (*

    Parser for operations in brackets

    *)
  10. bracket_singles : dispatch -> string -> unit Ast.expression Angstrom.t;
    (*

    Parser for all single program units like func call, lambda functions, if/then/else expressions and for operations or singles in brackets.

    *)
  11. all_singles : dispatch -> string -> unit Ast.expression Angstrom.t;
    (*

    Parser for all single program units like func call, lambda functions, if/then/else expressions and for operations or singles in brackets and for arithmetical priority operations (\*, /, %).

    *)
  12. all_ops : dispatch -> string -> unit Ast.expression Angstrom.t;
    (*

    Parser for all expressions in MiniML except for let and let in expressions.

    *)
  13. parse_letin : dispatch -> string -> unit Ast.expression Angstrom.t;
    (*

    Parser for let in definitions

    *)
  14. parse_def : dispatch -> string -> unit Ast.statement Angstrom.t;
    (*

    Parser for let definitions

    *)
}
