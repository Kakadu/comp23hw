Simplified ast representation which is the result of lambda lifting stage. The difference with cc_expr is that there are no nested and anonymous functions

type ll_expr =
  1. | LLiteral of Ast.literal
  2. | LIdentifier of Ast.identifier
  3. | LUnaryOp of Ast.unary_op * ll_expr
  4. | LBinaryOp of Ast.binary_op * ll_expr * ll_expr
  5. | LApp of ll_expr * ll_expr
  6. | LIfThenElse of ll_expr * ll_expr * ll_expr
  7. | LLetIn of Ast.identifier * ll_expr * ll_expr
val pp_ll_expr : Ppx_deriving_runtime.Format.formatter -> ll_expr -> Ppx_deriving_runtime.unit
val show_ll_expr : ll_expr -> Ppx_deriving_runtime.string
type ll_binding =
  1. | LVal of Ast.identifier * ll_expr
  2. | LFun of Ast.identifier * Ast.identifier list * ll_expr
val pp_ll_binding : Ppx_deriving_runtime.Format.formatter -> ll_binding -> Ppx_deriving_runtime.unit
val show_ll_binding : ll_binding -> Ppx_deriving_runtime.string
