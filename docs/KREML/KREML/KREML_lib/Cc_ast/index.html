Simplified ast representation which is the result of closure conversion stage. The difference is that there's 1 binding in let_in expr and uncurried lambda functions

type cc_expr =
  1. | CLiteral of Ast.literal
  2. | CIdentifier of Ast.identifier
  3. | CUnaryOp of Ast.unary_op * cc_expr
  4. | CBinaryOp of Ast.binary_op * cc_expr * cc_expr
  5. | CApp of cc_expr * cc_expr
  6. | CAbs of Ast.identifier list * cc_expr
  7. | CIfThenElse of cc_expr * cc_expr * cc_expr
  8. | CLetIn of cc_binding * cc_expr
and cc_binding =
  1. | CVal of Ast.identifier * cc_expr
  2. | CFun of Ast.identifier * Ast.identifier list * cc_expr
val pp_cc_expr : Ppx_deriving_runtime.Format.formatter -> cc_expr -> Ppx_deriving_runtime.unit
val show_cc_expr : cc_expr -> Ppx_deriving_runtime.string
val pp_cc_binding : Ppx_deriving_runtime.Format.formatter -> cc_binding -> Ppx_deriving_runtime.unit
val show_cc_binding : cc_binding -> Ppx_deriving_runtime.string
