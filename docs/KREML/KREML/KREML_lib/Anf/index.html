type immexpr =
  1. | ImmInt of int
  2. | ImmBool of bool
  3. | ImmIdentifier of string
type cexpr =
  1. | CImmExpr of immexpr
  2. | CUnaryOp of Ast.unary_op * immexpr
  3. | CBinaryOp of Ast.binary_op * immexpr * immexpr
  4. | CApp of immexpr * immexpr
  5. | CIfThenElse of immexpr * aexpr * aexpr
and aexpr =
  1. | ALet of string * cexpr * aexpr
  2. | ACExpr of cexpr
type abinding =
  1. | AFun of Ast.identifier * Ast.identifier list * aexpr
val show_abinding : abinding -> string
