type ilvalue =
  1. | DLvAny
    (*

    lvalue _

    *)
  2. | DLvUnit
    (*

    lvalue ()

    *)
  3. | DLvValue of int
    (*

    lvalue like a

    *)
  4. | DLvTuple of ilvalue list
    (*

    lvalue like (lvalue, lvalue, ...)

    *)

  1. | DApply of index_expr * index_expr

    index expr like a/b/c
