Module Parser_ast.ParserAst

type name = string
type rec_flag =
  1. | Rec
    (*

    Recursive

    *)
  2. | NoRec
    (*

    Not recursive

    *)
type lvalue =
  1. | LvAny
    (*

    Value _

    *)
  2. | LvUnit
    (*

    Value ()

    *)
  3. | LvValue of name
    (*

    Values like a

    *)
  4. | LvTuple of loc_lvalue list
    (*

    Values like (a, 3)

    *)
and loc_lvalue = lvalue Position.Position.position
type literal =
  1. | LInt of int
    (*

    Literals like 3

    *)
  2. | LFloat of float
    (*

    Literals like 3.2

    *)
  3. | LString of string
    (*

    Literals like "abc"

    *)
  4. | LBool of bool
    (*

    Literals true/false

    *)
  5. | LUnit
    (*

    Literal ()

    *)
type loc_literal = literal Position.Position.position
type let_binding = {
  1. rec_f : rec_flag;
  2. l_v : loc_lvalue;
  3. args : loc_lvalue list;
  4. body : loc_let_body;
}
and loc_let_binding = let_binding Position.Position.position
and let_body = {
  1. lets : loc_let_binding list;
  2. expr : loc_expr;
}
and loc_let_body = let_body Position.Position.position
and expr =
  1. | EApply of loc_expr * loc_expr
    (*

    e e

    *)
  2. | ELiteral of loc_literal
    (*

    "among" or 90

    *)
  3. | EValue of name
    (*

    a

    *)
  4. | EFun of {
    1. lvalue : loc_lvalue;
    2. body : loc_let_body;
    }
    (*

    fun x -> fun y -> ... fun x y -> ... represented as EFun x 

    ], EFun y ...}
    *)
  5. | ETuple of loc_expr list
    (*

    (a, b, c)

    *)
  6. | EIfElse of {
    1. cond : loc_expr;
    2. t_body : loc_expr;
    3. f_body : loc_expr;
    }
    (*

    if else statement

    *)
and loc_expr = expr Position.Position.position
type program = loc_let_binding list
val show_name : name -> string
val show_rec_flag : rec_flag -> string
val show_lvalue : lvalue -> string
val show_literal : literal -> string
val show_let_binding : let_binding -> string
val show_let_body : let_body -> string
val show_expr : expr -> string
val show_program : program -> string
val show_loc_lvalue : loc_lvalue -> string
val show_loc_literal : loc_literal -> string
val show_loc_let_binding : loc_let_binding -> string
val show_loc_let_body : loc_let_body -> string
val show_loc_expr : loc_expr -> string

val pp_lvalue : Stdlib.Format.formatter -> lvalue -> unit
val pp_literal : Stdlib.Format.formatter -> literal -> unit
val pp_let_binding : Stdlib.Format.formatter -> let_binding -> unit
val pp_let_body : Stdlib.Format.formatter -> let_body -> unit
val pp_expr : Stdlib.Format.formatter -> expr -> unit
val pp_program : Stdlib.Format.formatter -> program -> unit
val pp_loc_lvalue : Stdlib.Format.formatter -> loc_lvalue -> unit
val pp_loc_literal : Stdlib.Format.formatter -> loc_literal -> unit
val pp_loc_let_binding : Stdlib.Format.formatter -> loc_let_binding -> unit
val pp_loc_let_body : Stdlib.Format.formatter -> loc_let_body -> unit
