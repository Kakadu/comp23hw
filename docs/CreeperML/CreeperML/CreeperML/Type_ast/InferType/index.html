Module Type_ast.InferType

type lvl = int
type ground_typ =
  1. | TInt
    (*

    int type

    *)
  2. | TString
    (*

    string type

    *)
  3. | TBool
    (*

    bool type

    *)
  4. | TUnit
    (*

    unit type

    *)
  5. | TFloat
    (*

    float type

    *)
type 'a lvls = {
  1. value : 'a;
  2. mutable old_lvl : lvl;
  3. mutable new_lvl : lvl;
}
type ty =
  1. | TArrow of typ * typ
    (*

    typ -> typ type

    *)
  2. | TTuple of typ list
    (*

    (typ, ...) type

    *)
  3. | TGround of ground_typ
    (*

    int/string/bool/unit/float type

    *)
  4. | TVar of tv Stdlib.ref
    (*

    'a type

    *)
and typ = ty lvls
and tv =
  1. | Unbound of Parser_ast.ParserAst.name * lvl
    (*

    unbound type var

    *)
  2. | Link of typ
    (*

    link to typ

    *)
type env = (Parser_ast.ParserAst.name * typ) list
val show_lvl : lvl -> string
val show_ground_typ : ground_typ -> string

val show_tv : tv -> string

val show_typ : typ -> string
val show_env : env -> string

val pp_ground_typ : Stdlib.Format.formatter -> ground_typ -> unit

val pp_lvls : (Stdlib.Format.formatter -> 'a -> unit) -> Stdlib.Format.formatter -> 'a lvls -> unit
val pp_typ : Stdlib.Format.formatter -> typ -> unit
val pp_env : Stdlib.Format.formatter -> env -> unit
