type ty =
  1. | TyArrow of ty * ty
    (*

    ty -> ty type

    *)
  2. | TyTuple of ty list
    (*

    (ty, ty, ..) type

    *)
  3. | TyGround of InferType.ground_typ
    (*

    ground type

    *)
  4. | TyVar of Parser_ast.ParserAst.name
    (*

    'a type

    *)
type ('a, 'b) typed = {
  1. value : 'a;
  2. typ : 'b;
}
type 'ty typ_lvalue = (Parser_ast.ParserAst.lvalue, 'ty) typed
type 'ty typ_let_binding = {
  1. rec_f : Parser_ast.ParserAst.rec_flag;
  2. l_v : 'ty typ_lvalue;
  3. body : 'ty typ_let_body;
}
and 'ty typ_let_body = {
  1. lets : 'ty typ_let_binding list;
  2. expr : 'ty typ_expr;
}
and 'ty t_expr =
  1. | TApply of 'ty typ_expr * 'ty typ_expr
    (*

    typed application (a b)

    *)
  2. | TLiteral of Parser_ast.ParserAst.literal
    (*

    typed literal (3; "123")

    *)
  3. | TValue of Parser_ast.ParserAst.name
    (*

    typed value (a; something)

    *)
  4. | TFun of 'ty tfun_body
    (*

    typed fun (fun arg -> ...)

    *)
  5. | TTuple of 'ty typ_expr list
    (*

    typed tuple ((3, b c, (1, 2)))

    *)
  6. | TIfElse of 'ty tif_else
    (*

    typed if-then-else (if a then b else c)

    *)
and 'ty tfun_body = {
  1. lvalue : 'ty typ_lvalue;
  2. b : 'ty typ_let_body;
}
and 'ty tif_else = {
  1. cond : 'ty typ_expr;
  2. t_body : 'ty typ_expr;

}
